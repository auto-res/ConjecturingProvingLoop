\[ A \cap B \subseteq \text{int}(\overline{\text{int}(A \cap B)}) \]

\[ A \cap B \subseteq \text{int}(\overline{\text{int}(A)}) \cap \text{int}(\overline{\text{int}(B)}) \]

\text{int}(\overline{\text{int}(A \cap B)}) \subseteq \text{int}(\overline{\text{int}(A)}) \cap \text{int}(\overline{\text{int}(B)})

\[
\text{int}(\overline{\text{int}(A)}) \cap \text{int}(\overline{\text{int}(B)}) \subseteq \text{int}(\overline{\text{int}(A \cap B)})
